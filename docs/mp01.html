# Define a helper to prettify column names for display:
#     - underscores → spaces, then Title Case

library(stringr)
format_titles <- function(df) {
  colnames(df) <- colnames(df) |>
    str_replace_all("_", " ") |>
    str_to_title()
  df
}


# Preview with pretty headers, first 20 rows
